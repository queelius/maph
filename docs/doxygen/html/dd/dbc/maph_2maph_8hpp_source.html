92 static std::unique_ptr<Maph> create(const std::string& path,
93 size_t num_slots) {
94 auto m = std::make_unique<Maph>();
95
96 size_t file_size = sizeof(Header) + (num_slots * sizeof(Slot));
97
98 int fd = ::open(path.c_str(), O_RDWR | O_CREAT | O_TRUNC, 0666);
99 if (fd < 0) return nullptr;
100
101 if (ftruncate(fd, file_size) < 0) {
102 ::close(fd);
103 return nullptr;
104 }
105
106 void* mapped = mmap(nullptr, file_size, PROT_READ | PROT_WRITE,
107 MAP_SHARED, fd, 0);
108 if (mapped == MAP_FAILED) {
109 ::close(fd);
110 return nullptr;
111 }
112
113 // Initialize header
114 Header* header = new (mapped) Header();
115 header->num_slots = num_slots;
116
117 // Zero slots
118 Slot* slots = reinterpret_cast<Slot*>(
119 static_cast<char*>(mapped) + sizeof(Header));
120 memset(slots, 0, num_slots * sizeof(Slot));
121
122 m->fd_ = fd;
123 m->mapped_ = mapped;
124 m->file_size_ = file_size;
125 m->header_ = header;
126 m->slots_ = slots;
127 m->readonly_ = false;
128
129 // Default hash function (can be overridden)
130 m->hash_fn_ = [num_slots](const std::string& key) {
131 std::hash<std::string> h;
132 return h(key) % num_slots;
133 };
134
135 return m;
136 }

138 static std::unique_ptr<Maph> open(const std::string& path,
139 bool readonly = false) {
140 auto m = std::make_unique<Maph>();
141
142 int flags = readonly ? O_RDONLY : O_RDWR;
143 int fd = ::open(path.c_str(), flags);
144 if (fd < 0) return nullptr;
145
146 struct stat st;
147 if (fstat(fd, &st) < 0) {
148 ::close(fd);
149 return nullptr;
150 }
151
152 int prot = readonly ? PROT_READ : (PROT_READ | PROT_WRITE);
153 void* mapped = mmap(nullptr, st.st_size, prot, MAP_SHARED, fd, 0);
154 if (mapped == MAP_FAILED) {
155 ::close(fd);
156 return nullptr;
157 }
158
159 Header* header = static_cast<Header*>(mapped);
160 if (header->magic != 0x4D415048) {
161 munmap(mapped, st.st_size);
162 ::close(fd);
163 return nullptr;
164 }
165
166 m->fd_ = fd;
167 m->mapped_ = mapped;
168 m->file_size_ = st.st_size;
169 m->header_ = header;
170 m->slots_ = reinterpret_cast<Slot*>(
171 static_cast<char*>(mapped) + sizeof(Header));
172 m->readonly_ = readonly;
173
174 // Default hash
175 size_t num_slots = header->num_slots;
176 m->hash_fn_ = [num_slots](const std::string& key) {
177 std::hash<std::string> h;
178 return h(key) % num_slots;
179 };
180
181 return m;
182 }

186 // Get value for JSON key - direct memory access
187 std::string get(const std::string& json_key) const {
188 uint32_t index = hash_fn_(json_key);
189 if (index >= header_->num_slots) return "";
190
191 const Slot& slot = slots_[index];
192 if (slot.size == 0) return "";
193
194 // Optional: verify hash matches
195 // if (slot.hash != simple_hash(json_key)) return "";
196
197 return std::string(slot.data, slot.size);
198 }
199
200 // Set value - just memcpy
201 bool set(const std::string& json_key, const std::string& json_value) {
202 if (readonly_) return false;
203 if (json_value.size() > Slot::MAX_JSON_SIZE) return false;
204
205 uint32_t index = hash_fn_(json_key);
206 if (index >= header_->num_slots) return false;
207
208 Slot& slot = slots_[index];
209
210 // Lock-free update
211 uint32_t v = slot.version.fetch_add(1);
212
213 slot.size = json_value.size();
214 slot.hash = simple_hash(json_key);
215 memcpy(slot.data, json_value.data(), json_value.size());
216
217 slot.version.fetch_add(1);
218 header_->generation.fetch_add(1);
219
220 return true;
221 }
222
223 // Check existence - even faster
224 bool exists(const std::string& json_key) const {
225 uint32_t index = hash_fn_(json_key);
226 if (index >= header_->num_slots) return false;
227 return slots_[index].size > 0;
228 }
229
230 // Delete entry
231 bool remove(const std::string& json_key) {
232 if (readonly_) return false;
233
234 uint32_t index = hash_fn_(json_key);
235 if (index >= header_->num_slots) return false;
236
237 Slot& slot = slots_[index];
238 slot.version.fetch_add(1);
239 slot.size = 0;
240 slot.version.fetch_add(1);
241 header_->generation.fetch_add(1);
242
243 return true;
244 }

248 // Prefetch and batch get
249 std::vector<std::string> mget(const std::vector<std::string>& keys) const {
250 std::vector<std::string> results;
251 results.reserve(keys.size());
252
253 // Prefetch all slots into cache
254 for (const auto& key : keys) {
255 uint32_t index = hash_fn_(key);
256 if (index < header_->num_slots) {
257 __builtin_prefetch(&slots_[index], 0, 3);
258 }
259 }
260
261 // Now read (from cache)
262 for (const auto& key : keys) {
263 results.push_back(get(key));
264 }
265
266 return results;
267 }
