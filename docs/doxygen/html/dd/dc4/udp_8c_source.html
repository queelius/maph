105
106 LIBUS_SOCKET_DESCRIPTOR fd = bsd_create_udp_socket(host, port);
107 if (fd == LIBUS_SOCKET_ERROR) {
108 return 0;
109 }
110
111 /* If buf is 0 then create one here */
112 if (!buf) {
113 buf = us_create_udp_packet_buffer();
114 }
115
116 int ext_size = 0;
117 int fallthrough = 0;
118
119 struct us_poll_t *p = us_create_poll(loop, fallthrough, sizeof(struct us_internal_udp_t) + ext_size);
120 us_poll_init(p, fd, POLL_TYPE_CALLBACK);
121
122 struct us_internal_udp_t *cb = (struct us_internal_udp_t *) p;
123 cb->cb.loop = loop;
124 cb->cb.cb_expects_the_loop = 0;
125 cb->cb.leave_poll_ready = 1;
126
127 /* Get and store the port once */
128 struct bsd_addr_t tmp;
129 bsd_local_addr(fd, &tmp);
130 cb->port = bsd_addr_get_port(&tmp);
131
132 printf("The port of UDP is: %d\n", cb->port);
133
134 /* There is no udp socket context, only user data */
135 /* This should really be ext like everything else */
136 cb->user = user;
137
138 cb->data_cb = data_cb;
139 cb->receive_buf = buf;
140 cb->drain_cb = drain_cb;
141
142 cb->cb.cb = (void (*)(struct us_internal_callback_t *)) internal_on_udp_read;
143
144 us_poll_start((struct us_poll_t *) cb, cb->cb.loop, LIBUS_SOCKET_READABLE);
145
146 return (struct us_udp_socket_t *) cb;
147}
