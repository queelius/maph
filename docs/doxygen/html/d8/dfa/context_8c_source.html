33void us_listen_socket_close(int ssl, struct us_listen_socket_t *ls) {
34 /* us_listen_socket_t extends us_socket_t so we close in similar ways */
35 if (!us_socket_is_closed(0, &ls->s)) {
36 us_internal_socket_context_unlink_listen_socket(ls->s.context, ls);
37 us_poll_stop((struct us_poll_t *) &ls->s, ls->s.context->loop);
38 bsd_close_socket(us_poll_fd((struct us_poll_t *) &ls->s));
39
40 /* Link this socket to the close-list and let it be deleted after this iteration */
41 ls->s.next = ls->s.context->loop->data.closed_head;
42 ls->s.context->loop->data.closed_head = &ls->s;
43
44 /* Any socket with prev = context is marked as closed */
45 ls->s.prev = (struct us_socket_t *) ls->s.context;
46 }
47
48 /* We cannot immediately free a listen socket as we can be inside an accept loop */
49}

51void us_socket_context_close(int ssl, struct us_socket_context_t *context) {
52 /* Begin by closing all listen sockets */
53 struct us_listen_socket_t *ls = context->head_listen_sockets;
54 while (ls) {
55 struct us_listen_socket_t *nextLS = (struct us_listen_socket_t *) ls->s.next;
56 us_listen_socket_close(ssl, ls);
57 ls = nextLS;
58 }
59
60 /* Then close all regular sockets */
61 struct us_socket_t *s = context->head_sockets;
62 while (s) {
63 struct us_socket_t *nextS = s->next;
64 us_socket_close(ssl, s, 0, 0);
65 s = nextS;
66 }
67}

69void us_internal_socket_context_unlink_listen_socket(struct us_socket_context_t *context, struct us_listen_socket_t *ls) {
70 /* We have to properly update the iterator used to sweep sockets for timeouts */
71 if (ls == (struct us_listen_socket_t *) context->iterator) {
72 context->iterator = ls->s.next;
73 }
74
75 if (ls->s.prev == ls->s.next) {
76 context->head_listen_sockets = 0;
77 } else {
78 if (ls->s.prev) {
79 ls->s.prev->next = ls->s.next;
80 } else {
81 context->head_listen_sockets = (struct us_listen_socket_t *) ls->s.next;
82 }
83 if (ls->s.next) {
84 ls->s.next->prev = ls->s.prev;
85 }
86 }
87}

89void us_internal_socket_context_unlink_socket(struct us_socket_context_t *context, struct us_socket_t *s) {
90 /* We have to properly update the iterator used to sweep sockets for timeouts */
91 if (s == context->iterator) {
92 context->iterator = s->next;
93 }
94
95 if (s->prev == s->next) {
96 context->head_sockets = 0;
97 } else {
98 if (s->prev) {
99 s->prev->next = s->next;
100 } else {
101 context->head_sockets = s->next;
102 }
103 if (s->next) {
104 s->next->prev = s->prev;
105 }
106 }
107}

109/* We always add in the top, so we don't modify any s.next */
110void us_internal_socket_context_link_listen_socket(struct us_socket_context_t *context, struct us_listen_socket_t *ls) {
111 ls->s.context = context;
112 ls->s.next = (struct us_socket_t *) context->head_listen_sockets;
113 ls->s.prev = 0;
114 if (context->head_listen_sockets) {
115 context->head_listen_sockets->s.prev = &ls->s;
116 }
117 context->head_listen_sockets = ls;
118}

120/* We always add in the top, so we don't modify any s.next */
121void us_internal_socket_context_link_socket(struct us_socket_context_t *context, struct us_socket_t *s) {
122 s->context = context;
123 s->next = context->head_sockets;
124 s->prev = 0;
125 if (context->head_sockets) {
126 context->head_sockets->prev = s;
127 }
128 context->head_sockets = s;

202struct us_socket_context_t *us_create_socket_context(int ssl, struct us_loop_t *loop, int context_ext_size, struct us_socket_context_options_t options) {
203#ifndef LIBUS_NO_SSL
204 if (ssl) {
205 /* This function will call us, again, with SSL = false and a bigger ext_size */
206 return (struct us_socket_context_t *) us_internal_create_ssl_socket_context(loop, context_ext_size, options);
207 }
208#endif
209
210 /* This path is taken once either way - always BEFORE whatever SSL may do LATER.
211 * context_ext_size will however be modified larger in case of SSL, to hold SSL extensions */
212
213 struct us_socket_context_t *context = malloc(sizeof(struct us_socket_context_t) + context_ext_size);
214 context->loop = loop;
215 context->head_sockets = 0;
216 context->head_listen_sockets = 0;
217 context->iterator = 0;
218 context->next = 0;
219 context->is_low_prio = default_is_low_prio_handler;
220
221 /* Begin at 0 */
222 context->timestamp = 0;
223 context->long_timestamp = 0;
224 context->global_tick = 0;
225
226 us_internal_loop_link(loop, context);
227
228 /* If we are called from within SSL code, SSL code will make further changes to us */
229 return context;
230}

232void us_socket_context_free(int ssl, struct us_socket_context_t *context) {
233#ifndef LIBUS_NO_SSL
234 if (ssl) {
235 /* This function will call us again with SSL=false */
236 us_internal_ssl_socket_context_free((struct us_internal_ssl_socket_context_t *) context);
237 return;
238 }
239#endif
240
241 /* This path is taken once either way - always AFTER whatever SSL may do BEFORE.
242 * This is the opposite order compared to when creating the context - SSL code is cleaning up before non-SSL */
243
244 us_internal_loop_unlink(context->loop, context);
245 free(context);
246}

249#ifndef LIBUS_NO_SSL

252 }
253#endif
254
255 LIBUS_SOCKET_DESCRIPTOR listen_socket_fd = bsd_create_listen_socket(host, port, options);
256
257 if (listen_socket_fd == LIBUS_SOCKET_ERROR) {
258 return 0;
259 }
260
261 struct us_poll_t *p = us_create_poll(context->loop, 0, sizeof(struct us_listen_socket_t));
262 us_poll_init(p, listen_socket_fd, POLL_TYPE_SEMI_SOCKET);
263 us_poll_start(p, context->loop, LIBUS_SOCKET_READABLE);
264
265 struct us_listen_socket_t *ls = (struct us_listen_socket_t *) p;
266
267 ls->s.context = context;
268 ls->s.timeout = 255;
269 ls->s.long_timeout = 255;
270 ls->s.low_prio_state = 0;
271 ls->s.next = 0;
272 us_internal_socket_context_link_listen_socket(context, ls);
273
274 ls->socket_ext_size = socket_ext_size;
275
276 return ls;
277}

380/* Note: This will set timeout to 0 */
381struct us_socket_t *us_socket_context_adopt_socket(int ssl, struct us_socket_context_t *context, struct us_socket_t *s, int ext_size) {
382#ifndef LIBUS_NO_SSL
383 if (ssl) {
384 return (struct us_socket_t *) us_internal_ssl_socket_context_adopt_socket((struct us_internal_ssl_socket_context_t *) context, (struct us_internal_ssl_socket_t *) s, ext_size);
385 }
386#endif
387
388 /* Cannot adopt a closed socket */
389 if (us_socket_is_closed(ssl, s)) {
390 return s;
391 }
392
393 if (s->low_prio_state != 1) {
394 /* This properly updates the iterator if in on_timeout */
395 us_internal_socket_context_unlink_socket(s->context, s);
396 }
397
398 struct us_socket_t *new_s = (struct us_socket_t *) us_poll_resize(&s->p, s->context->loop, sizeof(struct us_socket_t) + ext_size);
399 new_s->timeout = 255;
400 new_s->long_timeout = 255;
401
402 if (new_s->low_prio_state == 1) {
403 /* update pointers in low-priority queue */
404 if (!new_s->prev) new_s->context->loop->data.low_prio_head = new_s;
405 else new_s->prev->next = new_s;
406
407 if (new_s->next) new_s->next->prev = new_s;
408 } else {
409 us_internal_socket_context_link_socket(context, new_s);
410 }
411
412 return new_s;
413}
