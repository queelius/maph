167 uint32_t v = version();

238 static Result compute(JsonView key, uint64_t num_slots) {
239 // FNV-1a hash
240 uint32_t h = 2166136261u;
241 for (unsigned char c : key) {
242 h ^= c;
243 h *= 16777619u;
244 }
245 if (h == 0) h = 1; // Never return 0 (means empty)
246
247 // Simple modulo for index (perfect hash would go here)
248 uint32_t idx = h % static_cast<uint32_t>(num_slots);
249
250 return {h, idx};
251 }
252
253#ifdef __AVX2__
254 // SIMD batch hash for up to 8 keys
255 static void compute_batch_avx2(const JsonView* keys, size_t count,
256 uint64_t num_slots, Result* results) {
257 // Process 8 hashes in parallel using AVX2
258 const __m256i fnv_prime = _mm256_set1_epi32(16777619u);
259 const __m256i fnv_offset = _mm256_set1_epi32(2166136261u);
260
261 size_t i = 0;
262 for (; i + 8 <= count; i += 8) {
263 __m256i hashes = fnv_offset;
264
265 // Find minimum length for this batch
266 size_t min_len = keys[i].size();
267 for (size_t j = 1; j < 8; ++j) {
268 min_len = std::min(min_len, keys[i + j].size());
269 }
270
271 // Process common prefix with SIMD
272 for (size_t pos = 0; pos < min_len; ++pos) {
273 __m256i chars = _mm256_setr_epi32(
274 keys[i + 0][pos], keys[i + 1][pos],
275 keys[i + 2][pos], keys[i + 3][pos],
276 keys[i + 4][pos], keys[i + 5][pos],
277 keys[i + 6][pos], keys[i + 7][pos]
278 );
279
280 hashes = _mm256_xor_si256(hashes, chars);
281 hashes = _mm256_mullo_epi32(hashes, fnv_prime);
282 }
283
284 // Extract results and finish tails
285 alignas(32) uint32_t hash_array[8];
286 _mm256_store_si256((__m256i*)hash_array, hashes);
287
288 for (size_t j = 0; j < 8; ++j) {
289 uint32_t h = hash_array[j];
290 // Finish the tail for keys longer than min_len
291 for (size_t pos = min_len; pos < keys[i + j].size(); ++pos) {
292 h ^= static_cast<unsigned char>(keys[i + j][pos]);
293 h *= 16777619u;
294 }
295 if (h == 0) h = 1;
296 results[i + j] = {h, h % static_cast<uint32_t>(num_slots)};
297 }
298 }
299
300 // Process remaining keys
301 for (; i < count; ++i) {
302 results[i] = compute(keys[i], num_slots);
303 }
304 }
305#endif
306
307 // Batch hash with automatic SIMD detection
308 static void compute_batch(const std::vector<JsonView>& keys,
309 uint64_t num_slots,
310 std::vector<Result>& results) {
311 results.resize(keys.size());
312
313#ifdef __AVX2__
314 if (__builtin_cpu_supports("avx2")) {
315 compute_batch_avx2(keys.data(), keys.size(), num_slots, results.data());
316 return;
317 }
318#endif
319
320 // Fallback to scalar
321 for (size_t i = 0; i < keys.size(); ++i) {
322 results[i] = compute(keys[i], num_slots);
323 }
324 }

387 static std::unique_ptr<Maph> create(const std::string& path,
388 uint64_t total_slots,
389 uint64_t static_slots = 0) {
390 auto m = std::make_unique<Maph>();
391
392 size_t file_size = sizeof(Header) + (total_slots * sizeof(Slot));
393
394 int fd = ::open(path.c_str(), O_RDWR | O_CREAT | O_TRUNC, 0666);
395 if (fd < 0) return nullptr;
396
397 if (::ftruncate(fd, file_size) < 0) {
398 ::close(fd);
399 return nullptr;
400 }
401
402 void* mapped = ::mmap(nullptr, file_size, PROT_READ | PROT_WRITE,
403 MAP_SHARED, fd, 0);
404 if (mapped == MAP_FAILED) {
405 ::close(fd);
406 return nullptr;
407 }
408
409 // Initialize
410 auto* header = new (mapped) Header();
411 header->total_slots = total_slots;
412 // Default: 80% static, 20% dynamic for collision handling
413 header->static_slots = (static_slots > 0) ? static_slots :
414 static_cast<uint64_t>(total_slots * DEFAULT_STATIC_RATIO);
415
416 auto* slots = reinterpret_cast<Slot*>(
417 static_cast<char*>(mapped) + sizeof(Header));
418 std::memset(slots, 0, total_slots * sizeof(Slot));
419
420 m->fd_ = fd;
421 m->mapped_ = mapped;
422 m->file_size_ = file_size;
423 m->header_ = header;
424 m->slots_ = slots;
425
426 return m;

443 auto m = std::make_unique<Maph>();
444
445 int flags = readonly ? O_RDONLY : O_RDWR;
446 int fd = ::open(path.c_str(), flags);
447 if (fd < 0) return nullptr;
448
449 struct stat st;
450 if (::fstat(fd, &st) < 0) {
451 ::close(fd);
452 return nullptr;
453 }
454
455 int prot = readonly ? PROT_READ : (PROT_READ | PROT_WRITE);
456 void* mapped = ::mmap(nullptr, st.st_size, prot, MAP_SHARED, fd, 0);
457 if (mapped == MAP_FAILED) {

464 ::munmap(mapped, st.st_size);

498 std::optional<JsonView> get(JsonView key) const {
499 auto [hash, index] = Hash::compute(key, header_->total_slots);
500
501 // For static range, only check direct slot
502 if (index < header_->static_slots) {
503 const Slot& slot = slots_[index];
504 if (slot.hash() == hash) {
505 return slot.view();
506 }
507 return std::nullopt;
508 }
509
510 // For dynamic range, check with linear probing
511 for (size_t i = 0; i < MAX_PROBE_DISTANCE; ++i) {
512 uint32_t idx = (index + i) % header_->total_slots;
513 if (idx < header_->static_slots) continue;
514
515 const Slot& s = slots_[idx];
516 if (s.empty()) break;
517 if (s.hash() == hash) {
518 return s.view();
519 }
520 }
521
522 return std::nullopt;
523 }
524
539 bool set(JsonView key, JsonView value) {
540 if (readonly_ || value.size() > Slot::MAX_SIZE) return false;
541
542 auto [hash, index] = Hash::compute(key, header_->total_slots);
543
544 // Static range - direct set (always overwrites)
545 if (index < header_->static_slots) {
546 slots_[index].set(hash, value);
547 header_->generation.fetch_add(1, std::memory_order_relaxed);
548 return true;
549 }
550
551 // Dynamic range - linear probe for existing key or empty slot
552 for (size_t i = 0; i < MAX_PROBE_DISTANCE; ++i) {
553 uint32_t idx = (index + i) % header_->total_slots;
554 if (idx < header_->static_slots) continue;
555
556 Slot& slot = slots_[idx];
557 // Set if empty or if it's the same key (update)
558 if (slot.empty() || slot.hash() == hash) {
559 slot.set(hash, value);
560 header_->generation.fetch_add(1, std::memory_order_relaxed);
561 return true;
562 }
563 }
564
565 return false; // Table full in probe region
566 }

579 bool remove(JsonView key) {

583
584 // For static range, only check direct slot

765 template<typename Callback>
766 void parallel_mget(const std::vector<JsonView>& keys,
767 Callback&& cb,
768 size_t thread_count = 0) const {
769 if (thread_count == 0) {
770 thread_count = std::thread::hardware_concurrency();
771 }
772
773 if (keys.size() < thread_count * 10) {
774 // Not worth parallelizing
775 mget(keys, std::forward<Callback>(cb));
776 return;
777 }
778
779 const size_t chunk_size = (keys.size() + thread_count - 1) / thread_count;
780 std::vector<std::future<void>> futures;
781
782 for (size_t t = 0; t < thread_count; ++t) {
783 size_t start = t * chunk_size;
784 size_t end = std::min(start + chunk_size, keys.size());
785
786 futures.emplace_back(std::async(std::launch::async,
787 [this, &keys, &cb, start, end]() {
788 // Prefetch for this chunk
789 for (size_t i = start; i < end; ++i) {
790 auto [_, index] = Hash::compute(keys[i], header_->total_slots);
791 __builtin_prefetch(&slots_[index], 0, 3);
792 }
793
794 // Process chunk
795 for (size_t i = start; i < end; ++i) {
796 if (auto value = get(keys[i])) {
797 cb(keys[i], *value);
798 }
799 }
800 }));
801 }
802
803 for (auto& f : futures) {
804 f.wait();
805 }
806 }
