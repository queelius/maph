40void us_socket_remote_address(int ssl, struct us_socket_t *s, char *buf, int *length) {
41 struct bsd_addr_t addr;
42 if (bsd_remote_addr(us_poll_fd(&s->p), &addr) || *length < bsd_addr_get_ip_length(&addr)) {
43 *length = 0;
44 } else {
45 *length = bsd_addr_get_ip_length(&addr);
46 memcpy(buf, bsd_addr_get_ip(&addr), *length);
47 }
48}

104/* Same as above but emits on_close */
105struct us_socket_t *us_socket_close(int ssl, struct us_socket_t *s, int code, void *reason) {
106 if (!us_socket_is_closed(0, s)) {
107 if (s->low_prio_state == 1) {
108 /* Unlink this socket from the low-priority queue */
109 if (!s->prev) s->context->loop->data.low_prio_head = s->next;
110 else s->prev->next = s->next;
111
112 if (s->next) s->next->prev = s->prev;
113
114 s->prev = 0;
115 s->next = 0;
116 s->low_prio_state = 0;
117 } else {
118 us_internal_socket_context_unlink_socket(s->context, s);
119 }
120 us_poll_stop((struct us_poll_t *) s, s->context->loop);
121 bsd_close_socket(us_poll_fd((struct us_poll_t *) s));
122
123 /* Link this socket to the close-list and let it be deleted after this iteration */
124 s->next = s->context->loop->data.closed_head;
125 s->context->loop->data.closed_head = s;
126
127 /* Any socket with prev = context is marked as closed */
128 s->prev = (struct us_socket_t *) s->context;
129
130 return s->context->on_close(s, code, reason);
131 }
132 return s;
133}

147int us_socket_write(int ssl, struct us_socket_t *s, const char *data, int length, int msg_more) {
148#ifndef LIBUS_NO_SSL
149 if (ssl) {
150 return us_internal_ssl_socket_write((struct us_internal_ssl_socket_t *) s, data, length, msg_more);
151 }
152#endif
153
154 if (us_socket_is_closed(ssl, s) || us_socket_is_shut_down(ssl, s)) {
155 return 0;
156 }
157
158 int written = bsd_send(us_poll_fd(&s->p), data, length, msg_more);
159 if (written != length) {
160 s->context->loop->data.last_write_failed = 1;
161 us_poll_change(&s->p, s->context->loop, LIBUS_SOCKET_READABLE | LIBUS_SOCKET_WRITABLE);
162 }
163
164 return written < 0 ? 0 : written;
165}
