19// Slot stores the hash that perfect hash already computed
20struct alignas(64) FastSlot {
21 std::atomic<uint64_t> hash_and_version{0}; // High 32: hash, Low 32: version
22 uint32_t size{0};
23 char value[500]{};
24
25 static constexpr uint64_t EMPTY = 0;
26
27 // Extract hash and version from packed field (no branches!)
28 uint32_t get_hash() const {
29 return hash_and_version.load() >> 32;
30 }
31
32 uint32_t get_version() const {
33 return hash_and_version.load() & 0xFFFFFFFF;
34 }
35
36 void set_hash_version(uint32_t hash, uint32_t version) {
37 hash_and_version.store((uint64_t(hash) << 32) | version);
38 }
39
40 bool is_empty() const {
41 return hash_and_version.load() < (1ULL << 32); // Hash part is 0
42 }
43};

51template<typename Key>
52struct PerfectHashWithHash {
53 struct Result {
54 uint32_t hash; // The hash we computed
55 uint32_t index; // The perfect hash index
56 };
57
58 Result operator()(const Key& key, size_t num_slots) const {
59 // Step 1: Compute hash ONCE
60 uint32_t h = compute_hash(key);
61
62 // Step 2: Perfect hash function (simplified)
63 // In practice, use RecSplit or similar
64 uint32_t idx = (h * 0x9E3779B9) % num_slots; // Golden ratio hash
65
66 return {h, idx};
67 }
68
69private:
70 static uint32_t compute_hash(const Key& key) {
71 // FNV-1a or xxHash - FAST hash
72 if constexpr (std::is_same_v<Key, std::string>) {
73 uint32_t h = 2166136261u;
74 for (char c : key) {
75 h ^= uint8_t(c);
76 h *= 16777619u;
77 }
78 return h;
79 } else {
80 return std::hash<Key>{}(key);
81 }
82 }
83};

96 static std::unique_ptr<FastMaph> create(const std::string& path,
97 size_t num_slots) {
98 auto m = std::make_unique<FastMaph>();
99
100 size_t file_size = num_slots * sizeof(FastSlot);
101
102 int fd = ::open(path.c_str(), O_RDWR | O_CREAT | O_TRUNC, 0666);
103 if (fd < 0) return nullptr;
104
105 if (ftruncate(fd, file_size) < 0) {
106 ::close(fd);
107 return nullptr;
108 }
109
110 void* mapped = mmap(nullptr, file_size, PROT_READ | PROT_WRITE,
111 MAP_SHARED | MAP_POPULATE, fd, 0); // MAP_POPULATE prefaults pages
112 if (mapped == MAP_FAILED) {
113 ::close(fd);
114 return nullptr;
115 }
116
117 // Zero out (OS might do this but be sure)
118 memset(mapped, 0, file_size);
119
120 ::close(fd); // Can close fd after mmap
121
122 m->mapped_ = mapped;
123 m->file_size_ = file_size;
124 m->slots_ = static_cast<FastSlot*>(mapped);
125 m->num_slots_ = num_slots;
126
127 return m;
128 }
