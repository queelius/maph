23// Enhanced slot with hash and optional expiration
24struct alignas(64) HybridSlot {
25 std::atomic<uint32_t> version{0}; // Lock-free updates
26 uint32_t key_hash{0}; // 0 = empty, else hash of key
27 uint32_t size{0}; // Value size
28 uint32_t expires_at{0}; // 0 = never, else unix timestamp
29 char value[496]{}; // JSON value (or key+value if needed)
30
31 static constexpr size_t MAX_VALUE_SIZE = 496;
32 static constexpr uint32_t EMPTY = 0;
33 static constexpr uint32_t TOMBSTONE = 0xFFFFFFFF; // Deleted slot
34
35 bool is_empty() const { return key_hash == EMPTY; }
36 bool is_deleted() const { return key_hash == TOMBSTONE; }
37 bool is_expired() const {
38 if (expires_at == 0) return false;
39 return expires_at < std::chrono::system_clock::to_time_t(
40 std::chrono::system_clock::now());
41 }
42 bool is_valid() const {
43 return !is_empty() && !is_deleted() && !is_expired();
44 }
45};

110 header->num_static = static_slots;
111 header->num_dynamic = 0;

113 HybridSlot* slots = reinterpret_cast<HybridSlot*>(
114 static_cast<char*>(mapped) + sizeof(HybridHeader));
115 memset(slots, 0, total_slots * sizeof(HybridSlot));

117 m->fd_ = fd;
118 m->mapped_ = mapped;
119 m->file_size_ = file_size;

121 m->slots_ = slots;

178 // ===== DYNAMIC PATH: Add any key with probing =====
179
180 std::string get(const std::string& key) const {
181 uint32_t hash = hash_key(key);
182
183 // Try static first (if in range)
184 if (header_->num_static > 0) {
185 uint32_t static_idx = perfect_hash_(hash);
186 if (static_idx < header_->num_static) {
187 const HybridSlot& slot = slots_[static_idx];
188 if (slot.key_hash == hash && slot.is_valid()) {
189 return std::string(slot.value, slot.size);
190 }
191 }
192 }
193
194 // Linear probe in dynamic range
195 uint32_t start = header_->num_static;
196 uint32_t end = header_->num_slots;
197
198 // Start probe position
199 uint32_t idx = start + (hash % (end - start));
200
201 for (uint32_t i = 0; i < 100; ++i) { // Max probe distance
202 const HybridSlot& slot = slots_[idx];
203
204 if (slot.is_empty()) {
205 return ""; // Not found
206 }
207
208 if (slot.key_hash == hash && slot.is_valid()) {
209 return std::string(slot.value, slot.size);
210 }
211
212 // Linear probe (could use quadratic or double hashing)
213 idx = start + ((idx - start + 1) % (end - start));
214 }
215
216 return ""; // Not found after max probes
217 }

318 Stats get_stats() const {
319 Stats s{};
320 s.total_slots = header_->num_slots;
321 s.static_slots = header_->num_static;
322 s.dynamic_slots = s.total_slots - s.static_slots;
323
324 for (size_t i = 0; i < header_->num_static; ++i) {
325 if (slots_[i].is_valid()) s.used_static++;
326 if (slots_[i].is_expired()) s.expired++;
327 }
328
329 for (size_t i = header_->num_static; i < header_->num_slots; ++i) {
330 if (slots_[i].is_valid()) s.used_dynamic++;
331 if (slots_[i].is_expired()) s.expired++;
332 }
333
334 s.load_factor = (double)(s.used_static + s.used_dynamic) / s.total_slots;
335
336 return s;
